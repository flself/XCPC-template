+ 倍增

```cpp
void dfs(int u, int fa) {
    pa[u][0] = fa; dep[u] = dep[fa] + 1;
    for (int i = 1; i < SP; ++i) pa[u][i] = pa[pa[u][i - 1]][i - 1];
    for (int& v: G[u]) {
        if (v == fa) continue;
        dfs(v, u);
    }
}

int lca(int u, int v) {
    if (dep[u] < dep[v]) swap(u, v);
    int t = dep[u] - dep[v];
    for (int i = 0; i < SP; ++i) if (t & (1 << i)) u = pa[u][i];
    for (int i = SP-1; i >= 0; --i) {
        int uu = pa[u][i], vv = pa[v][i];
        if (uu != vv) { u = uu; v = vv; }
    }
    return u == v ? u : pa[u][0];
}
```

```cpp
                if (to == f) continue;
                fa[to] = u, dep[to] = dep[u] + 1;
                predfs1(to, u);
                siz[u] += siz[to];
```

### tarjan

```cpp
const int N = 1e5 + 10;
std::vector<int> G[N];
int dfn[N], low[N], clk;
void init() { clk = 0; memset(dfn, 0, sizeof(dfn)); }

/// @brief cut point
void tarjan(int u, int fa) {
    low[u] = dfn[u] = ++clk;
    int cc = fa != -1;
    for (auto& v: G[u]) {
        if (v == fa) continue;
        if (!dfn[v]) {
            tarjan(v, u);
            low[u] = std::min(low[u], low[v]);
            if (low[v] >= dfn[u]) ++cc;
        } else low[u] = std::min(low[u], dfn[v]);
    }
    if (cc > 1) {
        // u is cut point
    }
}

/// @brief bridge
void tarjan(int u, int fa) {
    low[u] = dfn[u] = ++clk;
    int _fst = 0;
    for (auto& v: G[u]) {
        if (v == fa) continue;
        if (!dfn[v]) {
            tarjan(v, u);
            low[u] = std::min(low[u], low[v]);
            if (low[v] > dfn[u]) {
                // u-v is bridge
            }
        } else low[u] = std::min(low[u], dfn[v]);
    }
}


int bccno[N], bcc_cnt;
std::vector<int> bcc[N];
/// @brief bcc
void tarjan(int u) {
    static int stk[N], top;
    static bool ins[N];
    dfn[u] = low[u] = ++clk;
    stk[++top] = u, ins[u] = true;
    for (auto& v: G[u]) {
        if (!dfn[v]) {
            tarjan(v);
            low[u] = std::min(low[u], low[v]);
        } else if (ins[v]) low[u] = std::min(low[u], dfn[v]);
    }
    if (dfn[u] == low[u]) {
        while (true) {
            int x = stk[top--];
            ins[x] = false;
            bccno[x] = bcc_cnt;
            bcc[bcc_cnt].push_back(x);
            if (x == u) break;
        }
        ++bcc_cnt;
    }
}
```

```cpp
    int stk[N]; // 栈
    bool isl[N];

    std::vector<int> lst;   // 当前虚树中的点
    void buildvt(std::vector<int> pt, int rt = 0) {
        for (auto x: lst) vg[x].clear(), isl[x] = 0;
        lst.clear();
        sort(pt.begin(), pt.end(), [](int u, int v) {return idx[u] < idx[v];});

        int tp = 0;
        stk[++tp] = rt; lst.push_back(rt);
        for (auto u: pt) {
            if (u != rt) {
                int ll = lca(u, stk[tp]);
                if (ll != stk[tp]) {
                    while (idx[ll] < idx[stk[tp-1]]) 
                        vg[stk[tp-1]].push_back(stk[tp]), --tp;
                    vg[ll].push_back(stk[tp]);
                    if (idx[ll] > idx[stk[tp-1]]) stk[tp] = ll, lst.push_back(ll);
                    else --tp;
                }
                stk[++tp] = u, lst.push_back(u);
            }
            isl[u] = 1;
        }
        while (tp > 1) {
            vg[stk[tp-1]].push_back(stk[tp]);
            --tp;
        }
    }

    void add(int u, int v, int w = 0) {
        g[u].emplace_back(v, w);
        g[v].emplace_back(u, w);
    }
}
```

### 笛卡尔树

```cpp
std::vector<std::array<int, 2> > g;
int dikaer(std::vector<int> a) {
    std::stack<int> stk;
    g.resize(a.size());
    for (int i = 1; i < a.size(); ++i) {
        int las = -1;
        while (!stk.empty() && a[stk.top()] <= a[i]) las = stk.top(), stk.pop();
        if (~las) g[i][0] = las;
        if (!stk.empty()) g[stk.top()][1] = i;
        stk.push(i);
    }
    int rt = -1;
    while (!stk.empty()) rt = stk.top(), stk.pop();
    return rt;
}
```

## 网络流

### Dinic

```cpp
class Dinic{
    struct E{
        int to, cp;
        E(int t, int c): to(t), cp(c) {}
    };

    int n, m, s, t;
    std::vector<E> edges;
    std::vector<std::vector<int> > G;
    int *d, *cur;
    
    bool BFS() {
        memset(d, 0, sizeof(int) * (n));
        std::queue<int> Q;
        Q.push(s); d[s] = 1;
        while (!Q.empty()) {
            int x = Q.front(); Q.pop();
            for (auto &i: G[x]) {
                E &e = edges[i];
                if (!d[e.to] && e.cp > 0) {
                    d[e.to] = d[x] + 1;
                    Q.push(e.to);
                }
            }
        }
        return d[t];
    }

    int DFS(int u, int cp) {
        if (u == t || !cp) return cp;
        int tmp = cp, f;
        for (int& i = cur[u]; i < G[u].size(); ++i) {
            E& e = edges[G[u][i]];
            if (d[u] + 1 == d[e.to]) {
                f = DFS(e.to, std::min(cp, e.cp));
                e.cp -= f;
                edges[G[u][i] ^ 1].cp += f;
                cp -= f;
                if (!cp) break;
            }
        }
        return tmp - cp;
    }
    
public:
    Dinic(int nn): n(nn), G(nn), m(0) {d = new int[nn], cur = new int[nn];}
    ~Dinic() {delete[] d; delete[] cur;}

    void add(int u, int v, int cap) {
        edges.emplace_back(v, cap);
        edges.emplace_back(u, 0);
        G[u].push_back(m++);
        G[v].push_back(m++);
    }

    int go(int ss, int tt) {
        s = ss, t = tt;
        int flow = 0;
        while (BFS()) {
            memset(cur, 0, sizeof(int) * n);
            flow += DFS(s, INF);
        }
        return flow;
    }
};
```

### MinCostMaxFlow

```cpp
class Min_Cost_Max_Flow {
    struct edge {
        int to, cap, cost, rev;
        edge(int t = -1, int ca = 0, int co = 0, int r = -1): to(t), cap(ca), cost(co), rev(r) {} 
    };
    std::vector<edge> E;
    std::vector<std::vector<int> > G;
    bool *vis;
    int n, dist;

    int dfs(int v, int t, int &cost, int cf) {
        if (cf == 0) return 0;

        if (v == t) { cost += dist * cf; return cf; }

        vis[v] = true;
        int gf = cf;
        for (int i = 0; i < G[v].size(); ++i) {
            edge &e = E[G[v][i]];

            if (e.cost == 0 && !vis[e.to]) {
                int f = dfs(e.to, t, cost, std::min(cf, e.cap));
                cf -= f;
                e.cap -= f;
                E[e.rev].cap += f;
            }
        }
        return gf - cf;
    }

    bool label() {
        int cur = INF;

        for (int i = 0; i < n; ++i) if (vis[i])
            for (int j = 0; j < G[i].size(); ++j) {
            edge &e = E[G[i][j]];

            if (!vis[e.to] && e.cap > 0)
                cur = std::min(cur, e.cost);
        }

        if (cur == INF)
            return false;

        for (int i = 0; i < n; ++i) if (vis[i])
            for (int j = 0; j < G[i].size(); ++j) {
            edge &e = E[G[i][j]];
            e.cost -= cur;
            E[e.rev].cost += cur;
        }

        dist += cur;
        return true;
    }

public:
    Min_Cost_Max_Flow(int n): n(n), G(n) {vis = new bool[n];}
    ~Min_Cost_Max_Flow() {delete[] vis;}

    void add_edge(int u, int v, int cap, int cost) {
        int id = E.size();
        E.push_back(edge(v, cap, cost, id + 1));
        E.push_back(edge(u, 0, -cost, id));
        G[u].push_back(id);
        G[v].push_back(id + 1);
    }

    std::pair<int, int> go(int s, int t) {
        dist = 0;
        int flow = 0, cost = 0;
        int cf = 0;

        do
            do {
                memset(vis, false, sizeof(bool) * n);
                flow += cf = dfs(s, t, cost, INF);
            } while (cf > 0);

        while (label());
        return {flow, cost};
    }
};
```
